#include <iostream>

void stackExample() {
    int x = 10; // Переменная 'x' выделена на стеке
    std::cout << "Stack x: " << x << std::endl;
    // Память для 'x' автоматически освободится здесь при выходе из функции
}

void heapExample() {
    int* y = new int(20); // Память для int выделена в куче, 'y' (указатель) - на стеке
    std::cout << "Heap *y: " << *y << std::endl;

    // Мы ДОЛЖНЫ освободить память сами!
    delete y; // Освобождаем память в куче
    // Важно: сам указатель 'y' (переменная на стеке) уничтожится автоматически.
}

int main() {
    stackExample(); // Работа со стеком
    heapExample();  // Работа с кучей

    // Пример массива в куче
    int size;
    std::cout << "Enter array size: ";
    std::cin >> size; // Размер неизвестен на этапе компиляции

    int* dynamicArray = new int[size]; // Выделяем массив в куче
    for(int i = 0; i < size; ++i) {
        dynamicArray[i] = i * i;
    }
    // ... работа с массивом ...
    delete[] dynamicArray; // ОСВОБОЖДАЕМ массив! Без этого - утечка памяти.

    return 0;
}
